const express = require('express'); const mongoose = require('mongoose'); const bodyParser = require('body-parser'); const app = express(); const port = 3000; app.use(bodyParser.json()); mongoose.connect('mongodb://localhost:27017/mydatabase', { useNewUrlParser: true, useUnifiedTopology: true }); const userSchema = new mongoose.Schema({ username: String, password: String, }); const User = mongoose.model('User', userSchema); app.post('/register', async (req, res) => { const { username, password } = req.body; try { const existingUser = await User.findOne({ username }); if (existingUser) { res.status(400).send('Brugernavnet eksisterer allerede. Vælg et andet.'); } else { const newUser = new User({ username, password }); await newUser.save(); res.status(200).send('Bruger oprettet.'); } } catch (error) { res.status(500).send('Fejl ved oprettelse af bruger.'); } }); app.get('/users', async (req, res) => { try { const users = await User.find({}, 'username'); res.status(200).json(users); } catch (error) { res.status(500).send('Fejl ved hentning af brugere.'); } }); app.listen(port, () => { console.log(`Server kører på http://localhost:${port}`); });
